    # 模块引用

    var math = require("math");

// math.js
exports.add = function () {
  var sum = 0,
    i = 0,
    args = arguments,
    l = args.length;
  while (i < l) {
    sum += args[i++];
  }
  return sum;
};

// program.js
var math = require("math");
exports.increment = function (val) {
  return math.add(val, 1);
};
